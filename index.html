const slots = { A: { vid: document.getElementById('video-A'), cvs: document.getElementById('canvas-A'), ctx: document.getElementById('canvas-A').getContext('2d'), ovl: document.getElementById('overlay-A'), fill: document.getElementById('fill-A'), prog: document.getElementById('progress-A'), lastRes: null, baseline: null, lastElbow: 0, isLock: false, processing: false }, B: { vid: document.getElementById('video-B'), cvs: document.getElementById('canvas-B'), ctx: document.getElementById('canvas-B').getContext('2d'), ovl: document.getElementById('overlay-B'), fill: document.getElementById('fill-B'), prog: document.getElementById('progress-B'), lastRes: null, baseline: null, lastElbow: 0, isLock: false, processing: false } }; const vLive = document.getElementById('video-live'); const pose = new Pose({ locateFile: (file) => `https://cdn.jsdelivr.net/npm/@mediapipe/pose/${file}` }); pose.setOptions({ modelComplexity: 1, smoothLandmarks: true }); let camera = null, isLive = false, globalAiBusy = false, facing = "environment"; const RELEASE_THRESHOLD = 25; // 關鍵修正：使用回調分發結果給正確的 Slot pose.onResults((res) => { if (res.image) { // 透過比對來源或是標記來判斷是哪一個 Slot 的結果 const id = res.image.id === "video-B" ? 'B' : 'A'; slots[id].lastRes = res; processArcheryLogic(id, res.poseLandmarks); slots[id].processing = false; } globalAiBusy = false; }); function getAng(p1, p2, p3) { if(!p1 || !p2 || !p3) return 0; const r = Math.atan2(p3.y-p2.y, p3.x-p2.x) - Math.atan2(p1.y-p2.y, p1.x-p2.x); let a = Math.abs(r * 180 / Math.PI); return Math.round(a > 180 ? 360 - a : a); } function processArcheryLogic(id, lm) { if (!lm) return; const isR = lm[14].y < lm[13].y; const dS = isR?12:11, dE = isR?14:13, dW = isR?16:15; const pS = isR?11:12, pE = isR?13:14, pH = isR?23:24; const elbow = getAng(lm[dS], lm[dE], lm[dW]); const fSh = getAng(lm[pE], lm[pS], lm[pH]); const bSh = getAng(lm[dE], lm[dS], lm[isR?24:23]); const torso = getAng({x:(lm[11].x+lm[12].x)/2, y:0}, {x:(lm[11].x+lm[12].x)/2, y:(lm[11].y+lm[12].y)/2}, {x:(lm[23].x+lm[24].x)/2, y:(lm[23].y+lm[24].y)/2}); document.getElementById(`elbow-${id}`).innerText = elbow + "°"; document.getElementById(`f-shoulder-${id}`).innerText = fSh + "°"; document.getElementById(`b-shoulder-${id}`).innerText = bSh + "°"; document.getElementById(`torso-${id}`).innerText = torso + "°"; if (slots[id].baseline && slots[id].lastElbow !== 0 && !slots[id].isLock) { const delta = Math.abs(elbow - slots[id].lastElbow); if (delta > RELEASE_THRESHOLD) { triggerRelease(id, elbow, fSh); } } slots[id].lastElbow = elbow; if (slots[id].baseline) { const drift = Math.abs(elbow - slots[id].baseline.elb) + Math.abs(fSh - slots[id].baseline.fSh); const driftEl = document.getElementById(`drift-${id}`); driftEl.innerText = drift + "°"; driftEl.style.color = drift > 5 ? "#ff4444" : "#00FF00"; } } function triggerRelease(id, elbow, fSh) { slots[id].isLock = true; const slotEl = document.getElementById(`slot-${id}`); const logEl = document.getElementById(`release-${id}`); slotEl.classList.add('release-active'); logEl.innerText = `肘:${elbow}° 肩:${fSh}°`; logEl.style.color = "#FFCC00"; setTimeout(() => { slotEl.classList.remove('release-active'); slots[id].isLock = false; }, 3000); } async function handleFile(e, id) { if (id === 'A' && isLive) { isLive = false; if(camera) await camera.stop(); vLive.srcObject = null; } const file = e.target.files[0]; if (!file) return; const s = slots[id]; if (id === 'B') document.getElementById('slot-B').classList.remove('hidden'); s.ovl.style.display = 'flex'; s.vid.src = URL.createObjectURL(file); // 給影片加上 ID 以便在 onResults 中辨認 s.vid.id = `video-${id}`; await s.vid.load(); const dur = s.vid.duration; for (let t = 0; t <= dur; t += 0.5) { s.vid.currentTime = t; await new Promise(r => s.vid.onseeked = r); await pose.send({ image: s.vid }); s.fill.style.width = Math.round((t/dur)*100) + "%"; } s.ovl.style.display = 'none'; s.vid.currentTime = 0; s.vid.play(); } async function activateLiveMode() { slots.A.vid.src = ""; isLive = true; if (camera) await camera.stop(); vLive.id = "video-A"; // Live 模式佔用 A 槽 camera = new Camera(vLive, { onFrame: async () => { if (isLive && !slots.A.processing) { slots.A.processing = true; await pose.send({ image: vLive }); } }, facingMode: facing, width: 640, height: 480 }); camera.start(); } function toggleCamera() { facing = (facing === "environment") ? "user" : "environment"; if (isLive) activateLiveMode(); } function resetBaseline(id) { const res = slots[id].lastRes?.poseLandmarks; if (res) { const isR = res[14].y < res[13].y; slots[id].baseline = { elb: getAng(res[isR?12:11], res[isR?14:13], res[isR?16:15]), fSh: getAng(res[isR?13:14], res[isR?11:12], res[isR?23:24]) }; alert("基準點鎖定成功！"); } } async function manualSeek(id) { slots[id].vid.currentTime = (slots[id].prog.value / 100) * slots[id].vid.duration; if(!slots[id].processing) { slots[id].processing = true; await pose.send({ image: slots[id].vid }); } } function syncPlay() { if (isLive) { isLive = false; if(camera) camera.stop(); } ['A', 'B'].forEach(id => { if (slots[id].vid.src) { slots[id].vid.currentTime = 0; slots[id].vid.play(); } }); } // 核心繪製與排程迴圈 function draw() { ['A', 'B'].forEach(id => { const s = slots[id]; const src = (id === 'A' && isLive) ? vLive : s.vid; if (src && src.readyState >= 2) { s.cvs.width = src.videoWidth; s.cvs.height = src.videoHeight; s.ctx.drawImage(src, 0, 0); // 更新進度條 if (!src.paused && !src.ended && src.duration) { s.prog.value = (src.currentTime / src.duration) * 100; // 關鍵：如果該 Slot 沒在處理，且全域 AI 也沒在忙，才發送請求 if (!s.processing && !globalAiBusy) { s.processing = true; globalAiBusy = true; pose.send({ image: src }); } } // 繪製骨架 if (s.lastRes?.poseLandmarks) { drawConnectors(s.ctx, s.lastRes.poseLandmarks, POSE_CONNECTIONS, {color:'#00FF00', lineWidth:4}); drawLandmarks(s.ctx, s.lastRes.poseLandmarks, {color:'#FFF', radius:2}); } } }); requestAnimationFrame(draw); } window.onload = () => { draw(); };
